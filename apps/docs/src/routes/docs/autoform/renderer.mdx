  initialCode={`export default function Example() {

    autoformRendererUiSchema,
  )

  return (
    <AutoFormModelRenderer
      model={model}
      surface="inline"
      submitLabel="Create trip"
      closeOnSubmit={false}
    />
  )
}`}

  scope={{

    normalizeJsonSchemaWithUiSchema,
    autoformUiSchemaJsonSchema,

  }}
  initialCode={`export default function Example() {
  const normalized = normalizeJsonSchemaWithUiSchema(
    autoformUiSchemaJsonSchema,
    autoformUiSchemaUiSchema,
  )
  const adapter = adaptAutoFormModelToDialogWrapper(normalized)

  return (
    <DialogWrapper
      trigger={<button type="button">Open dialog wrapper</button>}
      schema={adapter.schema}
    />
  )
}`}
/>
