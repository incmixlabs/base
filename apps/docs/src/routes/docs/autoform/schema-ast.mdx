  const schema = {
    $id: "schema:signup",
    type: "object",
    required: ["accountType", "profile"],
    $defs: {
      profile: {
        type: "object",
        required: ["email"],
        properties: {
          email: { type: "string", format: "email" },
          locale: { type: "string", default: "en-US" },
        },
      },
    },
    properties: {
      accountType: {
        type: "string",
        const: "business",
        default: "business",
      },
      profile: { $ref: "#/$defs/profile" },
      companyName: { type: "string", minLength: 2 },
    },
    if: {
      properties: {
        accountType: { const: "business" },
      },
    },
    then: {
      required: ["companyName"],
    },
  }

  const ast = schemaToAst(schema)

  return (
    <div style={{ display: "grid", gap: 16 }}>
      <div style={{ padding: 16, border: "1px solid rgba(148, 163, 184, 0.35)", borderRadius: 16, background: "rgba(255,255,255,0.9)" }}>
        <div style={{ fontWeight: 700, marginBottom: 8 }}>Runtime schema</div>
        <pre style={{ margin: 0, overflow: "auto", fontSize: 12 }}>{JSON.stringify(schema, null, 2)}</pre>
      </div>
      <div style={{ padding: 16, border: "1px solid rgba(148, 163, 184, 0.35)", borderRadius: 16, background: "rgba(255,255,255,0.9)" }}>
        <div style={{ fontWeight: 700, marginBottom: 8 }}>Derived AST</div>
        <pre style={{ margin: 0, overflow: "auto", fontSize: 12 }}>{JSON.stringify(ast, null, 2)}</pre>
      </div>
    </div>
  )
}`}
